加载态 loading state
请求在飞时按钮转圈、禁用重复提交;成功/失败各给反馈。好体验的最低要求。

② service — 唯一发 fetch(沿用 agent-api 的 request<T> 写法):

// src/features/knowledge/services/knowledge-api.ts
const API_BASE = (
  process.env.NEXT_PUBLIC_API_BASE_URL ?? "http://localhost:8001"
).replace(/\/$/, "");

async function request<T>(path: string, init?: RequestInit): Promise<T> {
  const res = await fetch(`${API_BASE}${path}`, {
    headers: { "Content-Type": "application/json" }, ...init,
  });
  if (!res.ok) throw new Error(`Request failed (${res.status})`);  // MDN: 404 也要自己判断
  return (await res.json()) as T;
}

export function ingest(input: IngestInput) {
  return request<IngestResult>("/ingest", { method: "POST", body: JSON.stringify(input) });
}
export function ask(input: AskInput) {
  return request<AskResult>("/knowledge/ask", { method: "POST", body: JSON.stringify(input) });
}

③ hook — 管加载态,组件不碰网络:

// src/features/knowledge/hooks/use-knowledge.ts
"use client";
export function useKnowledge() {
  const [asking, setAsking] = useState(false);
  const [result, setResult] = useState<AskResult | null>(null);
  const [error, setError] = useState<string | null>(null);

  const submit = useCallback(async (question: string) => {
    if (!question.trim()) return;
    setAsking(true); setError(null); setResult(null);
    try { setResult(await ask({ question, k: 3 })); }
    catch (e) { setError(e instanceof Error ? e.message : "提问失败"); }
    finally { setAsking(false); }
  }, []);

  return { asking, result, error, submit };
}

Primary source · 先读这个

MDN · Using the Fetch API 点明关键坑:fetch() 不会因 404 而 reject,它照常兑现一个 Response—— 所以必须先查 response.ok 再读 body,正是你 service 里那行 if (!res.ok) throw。 分层规则参见工作区的 RESOURCES 与前端架构约定。
